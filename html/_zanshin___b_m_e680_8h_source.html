147  /*************************************************************************************************************
148  ** Declare the getData and putData methods as template functions. All device I/O is done through these two **
149  ** functions regardless of whether I2C, hardware SPI or software SPI is being used. The two functions are **
150  ** designed so that only the address and a variable are passed in and the functions determine the size of **
151  ** the parameter variable and reads or writes that many bytes. So if a read is called using a character **
152  ** array[10] then 10 bytes are read, if called with a int8 then only one byte is read. The return value, if **
153  ** used, is the number of bytes read or written. Since this is implemented by using template function **
154  ** definitions, they need to be defined in this header file rather than in the c++ program library file. **
155  *************************************************************************************************************/
156  template< typename T > uint8_t &getData(const uint8_t addr,T &value)
157  {
165  uint8_t* bytePtr = (uint8_t*)&value; // Pointer to structure beginning
166  static uint8_t structSize = sizeof(T); // Number of bytes in structure
167  if (_I2CAddress) // Using I2C if address is non-zero
168  { //
169  Wire.beginTransmission(_I2CAddress); // Address the I2C device
170  Wire.write(addr); // Send register address to read
171  Wire.endTransmission(); // Close transmission
172  Wire.requestFrom(_I2CAddress, sizeof(T)); // Request 1 byte of data
173  structSize = Wire.available(); // Use the actual number of bytes
174  for (uint8_t i=0;i<structSize;i++) *bytePtr++ = Wire.read(); // loop for each byte to be read
175  }
176  else
177  {
178  if (_sck==0) { // if sck is zero then hardware SPI
179  SPI.beginTransaction(SPISettings(SPI_HERZ,MSBFIRST,SPI_MODE0)); // Start the SPI transaction
180  digitalWrite(_cs, LOW); // Tell BME680 to listen up
181  SPI.transfer(addr | 0x80); // bit 7 is high, so read a byte
182  for (uint8_t i=0;i<structSize;i++) *bytePtr++ = SPI.transfer(0); // loop for each byte to be read
183  digitalWrite(_cs, HIGH); // Tell BME680 to stop listening
184  SPI.endTransaction(); // End the transaction
185  } else { // otherwise we are using soft SPI
186  int8_t i,j; // Loop variables
187  uint8_t reply; // return byte for soft SPI transfer
188  digitalWrite(_cs, LOW); // Tell BME680 to listen up
189  for (j=7; j>=0; j--) { // First send the address byte
190  digitalWrite(_sck, LOW); // set the clock signal
191  digitalWrite(_mosi, ((addr)|0x80)&(1<<j)); // set the MOSI pin state
192  digitalWrite(_sck, HIGH); // reset the clock signal
193  } // of for-next each bit
194  for (i=0;i<structSize;i++) { // Loop for each byte to read
195  reply = 0; // reset our return byte
196  for (j=7; j>=0; j--) { // Now read the data at that byte
197  reply <<= 1; // shift buffer one bit left
198  digitalWrite(_sck, LOW); // set and reset the clock signal
199  digitalWrite(_sck, HIGH); // pin to get the next MISO bit
200  if (digitalRead(_miso)) reply |= 1; // read the MISO bit, add to reply
201  } // of for-next each bit
202  *bytePtr++ = reply; // Add byte just read to return data
203  } // of for-next each byte to be read
204  digitalWrite(_cs, HIGH); // Tell BME680 to stop listening
205  } // of if-then-else we are using hardware SPI
206  } // of if-then-else we are using I2C
207  return(structSize);
208  } // of method getData()
