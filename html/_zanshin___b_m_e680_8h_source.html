251  uint8_t * bytePtr = (uint8_t *)&value; // Pointer to structure beginning
252  static uint8_t structSize = sizeof(T); // Number of bytes in structure
253  if (_I2CAddress) // Using I2C if address is non-zero
254  { //
255  Wire.beginTransmission(_I2CAddress); // Address the I2C device
256  Wire.write(addr); // Send register address to read
257  Wire.endTransmission(); // Close transmission
258  Wire.requestFrom(_I2CAddress, sizeof(T)); // Request 1 byte of data
259  structSize = Wire.available(); // Use the actual number of bytes
260  for (uint8_t i = 0; i < structSize; i++)
261  *bytePtr++ = Wire.read(); // loop for each byte to be read
262  } //
263  else //
264  { //
265  if (_sck == 0) // if sck is zero then hardware SPI
266  { //
267  SPI.beginTransaction(
268  SPISettings(SPI_HERZ, MSBFIRST, SPI_MODE0)); // Start the SPI transaction
269  digitalWrite(_cs, LOW); // Tell BME680 to listen up
270  SPI.transfer(addr | 0x80); // bit 7 is high, so read a byte
271  for (uint8_t i = 0; i < structSize; i++)
272  *bytePtr++ = SPI.transfer(0); // loop for each byte to be read
273  digitalWrite(_cs, HIGH); // Tell BME680 to stop listening
274  SPI.endTransaction(); // End the transaction
275  } else { // otherwise we are using soft SPI
276  int8_t i, j; // Loop variables
277  uint8_t reply; // return byte for soft SPI transfer
278  digitalWrite(_cs, LOW); // Tell BME680 to listen up
279  for (j = 7; j >= 0; j--) // First send the address byte
280  {
281  digitalWrite(_sck, LOW); // set the clock signal
282  digitalWrite(_mosi, ((addr) | 0x80) & (1 << j)); // set the MOSI pin state
283  digitalWrite(_sck, HIGH); // reset the clock signal
284  } // of for-next each bit
285  for (i = 0; i < structSize; i++) // Loop for each byte to read
286  { //
287  reply = 0; // reset our return byte
288  for (j = 7; j >= 0; j--) // Now read the data at that byte
289  { //
290  reply <<= 1; // shift buffer one bit left
291  digitalWrite(_sck, LOW); // set and reset the clock signal
292  digitalWrite(_sck, HIGH); // pin to get the next MISO bit
293  if (digitalRead(_miso)) reply |= 1; // read the MISO bit, add to reply
294  } // of for-next each bit

296  } // of for-next each byte to be read

298  } // of if-then-else we are using hardware SPI
