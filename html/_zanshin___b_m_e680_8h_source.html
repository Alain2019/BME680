147  Wire.write(addr); // Send register address to read

150  structSize = Wire.available(); // Use the actual number of bytes
151  for (uint8_t i=0;i<structSize;i++) *bytePtr++ = Wire.read(); // loop for each byte to be read
152  } else {

155  digitalWrite(_cs, LOW); // Tell BME680 to listen up
156  SPI.transfer(addr | 0x80); // bit 7 is high, so read a byte
157  for (uint8_t i=0;i<structSize;i++) *bytePtr++ = SPI.transfer(0); // loop for each byte to be read
158  digitalWrite(_cs, HIGH); // Tell BME680 to stop listening
159  SPI.endTransaction(); // End the transaction
160  } else { // otherwise we are using soft SPI
161  int8_t i,j; // Loop variables
162  uint8_t reply; // return byte for soft SPI transfer
163  digitalWrite(_cs, LOW); // Tell BME680 to listen up
164  for (j=7; j>=0; j--) { // First send the address byte
165  digitalWrite(_sck, LOW); // set the clock signal
166  digitalWrite(_mosi, ((addr)|0x80)&(1<<j)); // set the MOSI pin state
167  digitalWrite(_sck, HIGH); // reset the clock signal
168  } // of for-next each bit
169  for (i=0;i<structSize;i++) { // Loop for each byte to read
170  reply = 0; // reset our return byte
171  for (j=7; j>=0; j--) { // Now read the data at that byte
172  reply <<= 1; // shift buffer one bit left
173  digitalWrite(_sck, LOW); // set and reset the clock signal
174  digitalWrite(_sck, HIGH); // pin to get the next MISO bit
175  if (digitalRead(_miso)) reply |= 1; // read the MISO bit, add to reply
176  } // of for-next each bit
177  *bytePtr++ = reply; // Add byte just read to return data
178  } // of for-next each byte to be read
179  digitalWrite(_cs, HIGH); // Tell BME680 to stop listening
180  } // of if-then-else we are using hardware SPI
181  } // of if-then-else we are using I2C
